\chapter{Distribuciones discretas}
En este capítulo se presentan las funciones de probabilidad de dos distribuciones discretas. En el capítulo~\ref{cap:cont} (en la página~\pageref{cap:cont}) se presentan dos distribuciones continuas.

\section{La distribución binomial}
La función de probabilidad de una distribución binomial con parámetros $n$ y $p$, $X\sim B(n,p)$, tiene la siguiente expresión:
\begin{equation*}
P_X(k) = {n\choose k}p^k(1-p)^{n-k}, \quad k\in\{0, 1, \dotsc, n\}.
\end{equation*}
En la sección \ref{sec:pois} se habla de la distribución de Poisson.

\section{La distribución de Poisson\label{sec:pois}}
Si $X\sim Po(\lambda)$, entonces
\begin{equation*}
P_X(k) = \frac{\lambda^k}{k!}e^{-\lambda}, \quad k=0,1,2,\dotsc.
\end{equation*}
