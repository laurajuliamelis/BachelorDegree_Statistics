# Apartat a 
desv.tipus <- 10*sqrt(1/20+1/20)
curve(dnorm(x,mean=0,sd=desv.tipus),from=-10,to=20, col="grey")

lim <- qnorm(p = 0.05, mean = 0, sd = desv.tipus, lower.tail = FALSE)
abline(v=lim, lty=2)
curve(dnorm(x,mean=8,sd=desv.tipus),add=TRUE, from=-10,to=20, col="red")

# Error de tipus II (no rebutgem H0 quan en realitat es falsa)
pnorm(q = lim, mean = 8, sd = desv.tipus, lower.tail = TRUE)
